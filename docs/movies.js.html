const { movieService } = require('../services');

/**
 * getMovies Route: /api/movies
 * @async
 * @param {Object} req
 * @param {Object} res
 * @returns {JSON} returns a JSON file containing list of movies
 */
module.exports.getMovies = async (req, res) => {
  try {
    const { status, response } = await movieService.getMovies();
    return res.status(status).send(response);
  } catch (err) {
    return res.status(500).send(`Error: ${err}`);
  }
};

/**
 * getMovies Route: /api/movie/:movie-title
 * @async
 * @param {Object} req
 * @param {Object} res
 * @returns {JSON} returns a JSON file containing the requested movie
 */
module.exports.getMovie = async (req, res) => {
  try {
    const { title } = req.params;
    const { status, response } = await movieService.getMovie(title);
    return res.status(status).send(response);
  } catch (err) {
    return res.status(500).send(`Error: ${err}`);
  }
};
